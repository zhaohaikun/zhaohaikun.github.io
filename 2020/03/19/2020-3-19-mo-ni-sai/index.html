T5

我们要贪心，要让攻者尽可能赢，也就是让攻者把防者的所有牌给打掉。

我们显然有一个贪心策略是这样的：

每次拿 $2$ 队最强的牌
则有 $2$ 种情况

  • 如果攻者的牌 $\geq$ 防者的牌，就显然要让 $2$ 张牌同归于尽。
  • 如果攻者的牌 $<$ 防者的牌，就显然要让防者的牌 $-$ 攻者的牌的值。
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
template<typename T>inline void read(T &FF){
    T RR=1;FF=0;char CH=getchar();
    for(;!isdigit(CH);CH=getchar())if(CH=='-')RR=-1;
    for(;isdigit(CH);CH=getchar())FF=(FF<<1)+(FF<<3)+(CH^48);
    FF*=RR;
}
template<typename T>inline void write(T x){
    if(x<0)putchar('-'),x*=-1;
    if(x>9)write(x/10);
    putchar(x%10+48);
}
template<typename T>inline void writen(T x){
    write(x);
    puts("");
}
ll T,n,m,a[300010],b[300010];
bool cmp(ll x,ll y){
    return x>y;
}
bool work(){
    priority_queue<ll>q;
    sort(a+1,a+n+1,cmp);
    for(int i=1;i<=m;i++)q.push(b[i]);
    for(int i=1;i<=n;i++){
        ll x=q.top();
        q.pop();
        if(a[i]<x){
            x-=a[i];
            q.push(x);
        }
        if(q.empty())return true;
    }return false;
}
int main(){
    read(T);
    while(T--){
        read(n);
        for(int i=1;i<=n;i++)read(a[i]);
        read(m);
        for(int i=1;i<=m;i++)read(b[i]);
        if(work())puts("Apollo");
        else puts("Dionysus");
    }
    return 0;
}

T6

这道题还是很有水平的。

有这样一个结论：可能的数字为所有素数的平方。

首先，$n+\left \lceil \left \vert \sqrt{n+1} \right \vert \right \rceil=n+ \lfloor \vert \sqrt{n} \vert\rfloor+1$

  1. 对于平方数

    1. 对于素数的平方：假设是 $x$ 的平方

      $\therefore$ 约数和 $= x^2+x+1$，上面写的式子也是 $x^2+x+1$

      $\because$ 可以。

    2. 对于约数的平方：假设是 $x$ 的平方。

      $\therefore$ 约数和 $\geq x^2+1+x+\sum\limits_{i \mid x} x^2 \div i > x^2+x+1$

      $\because$ 不可以。

  2. 那么非平方数，假设它为 $n$，$x=\sqrt{n}$。

    1. 如果 $n$ 为素数：

      $\therefore n$ 的约数和为：$n+1$，上面的式子为 $n+x+1$。

      又 $\therefore$ $x \geq 1$。

      $\because$ 不可以。

    2. 如果 $n$ 为合数：

      $\therefore$ $n$ 的约数和一定会大于 $> x^2+x+1$。

      $\because$ 不可以。

    ll a,b,c;

void dfs(ll x,ll s,ll y,ll minn,ll z){
    if(x==n+1){
        ans=max(ans,minn*(C-work(s))*(p[z%(Len)]));
        return;
    }
    dfs(x+1,s,y,minn,z);
    dfs(x+1,s+y+a[x].c,y+a[x].c,min(minn,a[x].b),a[x].a^z);
}

    cout<<ans;
